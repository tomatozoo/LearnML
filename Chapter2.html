# z점수 계산하기
mean = np.mean(train_input, axis=0)
std = np.std(train_input, axis=0) # 행을 기준으로(axis=0) 계산
print(mean, std)
train_scaled = (train_input - mean) / std

# 변환된 데이터 시각화하기
new = ([25, 150] - mean) / std
plt.scatter(train_scaled[:, 0], train_scaled[:, 1])
plt.scatter(new[0], new[1], marker="^")
